/**
 * ArticleDetails component displays detailed information about an article.
 *
 * @component
 * @returns {JSX.Element} - Rendered component.
 */
function ArticleDetails() {
    const { ArticleDisplay } = useContext(ArticleContext);
    const article = ArticleDisplay;
    const [texte, setTexte] = useState(article.full_text);

    /**
     * Counts the number of words in a given text.
     *
     * @param {string} str - The text to count words in.
     * @returns {number} - The number of words in the text.
     */
    const WordCount = (str) => {
        let totalSoFar = 0;
        for (let i = 0; i < texte.length; i++)
            if (str[i] === " ") {
                totalSoFar++;
            }
        return totalSoFar;
    };

    let len = WordCount(texte);
    let len6 = (len - (len % 3)) / 6;

    /**
     * Displays a range of words from the given text.
     *
     * @param {string} text - The text to display words from.
     * @param {number} start - The starting index of the word range.
     * @param {number} end - The ending index of the word range.
     * @returns {string} - The selected words from the range.
     */
    const displayWordRange = (text, start, end) => {
        const words = text.split(" ");
        const selectedWords = words.slice(start, end).join(" ");
        return selectedWords;
    };

    return (
        <div className="w-full">
            {/* ... (existing code) */}
        </div>
    );
}

export default ArticleDetails;
